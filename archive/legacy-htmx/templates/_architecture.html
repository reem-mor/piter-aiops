  input={"text": question},
  retrieveAndGenerateConfiguration={
    "type": "KNOWLEDGE_BASE",
    "knowledgeBaseConfiguration": {
      "knowledgeBaseId": "{{ kb_id }}",
      "modelArn": "{{ model_label }}",
    },
  },
